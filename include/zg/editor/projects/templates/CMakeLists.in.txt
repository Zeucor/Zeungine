cmake_minimum_required(VERSION 3.22.1)
set(CMAKE_CXX_STANDARD 20)
project(@PROJECT_NAME@ VERSION 0.1.0)
find_package(Zeungine)
file(GLOB PROJECT_SOURCES "src/*.cpp" "src/*.h" "src/*.c" "src/*.hpp")
include_directories(include)
message(STATUS "Zeungine_INCLUDE_DIR: ${Zeungine_INCLUDE_DIR}")
include_directories(${Zeungine_INCLUDE_DIR})
add_library(@LIBRARY_NAME@ SHARED ${PROJECT_SOURCES})
target_link_libraries(@LIBRARY_NAME@ ${Zeungine_LIBRARIES})
